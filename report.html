Before using any of the face detectors, it is standard procedure to convert the images to grayscale. The detectMultiScale function executes the classifier stored in face_cascade and takes the grayscale image as a parameter.

In the above code, faces is a numpy array of detected faces, where each row corresponds to a detected face. # returns "True" if face is detected in image stored at img_path
def face_detector(img_path):
    img = cv2.imread(img_path)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    faces = face_cascade.detectMultiScale(gray)
    return len(faces) > 0

from PIL import Image
import torchvision.transforms as transforms

if use_cuda:
    VGG16 = VGG16.cuda()

def image_loader(loader, image_name):
    image = Image.open(image_name)
    image = loader(image).float()
    image = torch.tensor(image, requires_grad=False)
    image = image.unsqueeze(0)
    return image



def VGG16_predict(img_path):
    '''
    Use pre-trained VGG-16 model to obtain index corresponding to 
    predicted ImageNet class for image at specified path
    
    Args:
        img_path: path to an image
        
    Returns:
        Index corresponding to VGG-16 model's prediction
    '''
    #Put model on evaluation mode
    VGG16.eval()
    
    ## Load and pre-process an image from the given img_path
    normalize = transforms.Normalize(mean=[0.485, 0.456, 0.406],
                                     std=[0.229, 0.224, 0.225])
    data_transforms = transforms.Compose([
        transforms.Resize(256),
        transforms.CenterCrop(224),
        transforms.ToTensor(),
        normalize,
    ])
    
    loaded_image = image_loader(data_transforms, img_path)
    
    if use_cuda:
        loaded_image = loaded_image.cuda()
    
    
    ## Return the *index* of the predicted class for that image
    idx = np.argmax(VGG16(loaded_image).detach().cpu().numpy())
    
    return idx # predicted class index

import torch.nn as nn
import torch.nn.functional as F

# define the CNN architecture
class Net(nn.Module):
    ### TODO: choose an architecture, and complete the class
    def __init__(self):
        super(Net, self).__init__()
        #First convolution layer sees (224x224x3 image tensor)
        self.conv1 = nn.Conv2d(3, 16, 3, padding = 1)
        #Second convolution layer sees (112x112x16 image tensor)
        self.conv2 = nn.Conv2d(16, 32, 3, padding = 1)
        #Third convolution layer sees (56x56x32 image tensor)
        self.conv3 = nn.Conv2d(32, 64, 3, padding = 1)
        #Maxpool layer reduces the height and width by half
        self.pool = nn.MaxPool2d(2,2)
        #fully connected layer input = (64*28*28 -> 500)
        self.fc1 = nn.Linear(64*28*28, 500)
        #fully connected layer (500 -> 133)
        self.fc2 = nn.Linear(500,133)
        #dropout layer with probablity 25%
        self.dropout = nn.Dropout(0.25)
    
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = self.pool(F.relu(self.conv3(x)))
        #flatten image
        x = x.view(-1, 64*28*28)
        #dropout layer
        #x = self.dropout(x)
        # add 1st hidden layer, with relu activation function
        x = F.relu(self.fc1(x))
        # add dropout layer
        x = self.dropout(x)
        # add 2nd hidden layer, without relu activation function
        x = self.fc2(x)
        return x

Question 4: Outline the steps you took to get to your final CNN architecture and your reasoning at each step.

Answer:

I have used three convolution layers wherein the depth increases from 3 at the input for first layer layer to 64 at output of the last convolution layer. The filter size is 3x3 with a padding of 1, such that the convolution returns an image having the same height and width as the input image. As a first attempt, taking inspiration from the previous exercises, I explored this architecture. The first conv layer probably will identify simples features such as edges (vertical or horizontal lines). The second layer probably identifies slightly more complicated shapes such as circles, stripes, etc. The final conv layer probably identifies even more complicated features. Better results can be achieved using more conv layers but due to time limit I have explored this setup. After every conv layer I have included a maxpooling layer which reduces the height and width by half. A relu activation function is used to reduce the impact of the vanishing gradient problem. The output of the convolution layers is attached to a fully connected classifier structure involving 2 fully connected vanilla neural network architecture. I chose 2 fc layers at first, again taking inspiration from the exercises. This seems to do what is required for the basic dog breed classifier, although the better pretrained models such as VGG16 use just one more fc layer (total three). Between these fc layers i have added a dropout probablity of 25%. Again, the ReLU activation function is used to reduce the vanishing gradient problem. The final output has no activation function since the cross entropy loss function performs the log softmax by itself. Using this architecture I was able to produce a model with accuracy of about 14%.

def train(n_epochs, loaders, model, optimizer, criterion, use_cuda, save_path):
    """returns trained model"""
    # initialize tracker for minimum validation loss
    valid_loss_min = np.Inf 
    
    for epoch in range(1, n_epochs+1):
        # initialize variables to monitor training and validation loss
        train_loss = 0.0
        valid_loss = 0.0
        
        ###################
        # train the model #
        ###################
        model.train()
        for batch_idx, (data, target) in enumerate(loaders['train']):
            # move to GPU
            if use_cuda:
                data, target = data.cuda(), target.cuda()
            ## find the loss and update the model parameters accordingly
            ## record the average training loss, using something like
            ## train_loss = train_loss + ((1 / (batch_idx + 1)) * (loss.data - train_loss))
            
            #zero the gradients
            optimizer.zero_grad()
            # forward pass
            output = model(data)
            # calculate the batch loss
            loss = criterion(output, target)
            # backward pass: compute gradient of the loss with respect to model parameters
            loss.backward()
            # perform a single optimization step (parameter update)
            optimizer.step()
            #Update train loss
            train_loss = train_loss + ((1 / (batch_idx + 1)) * (loss.data - train_loss))
            
        ######################    
        # validate the model #
        ######################
        model.eval()
        for batch_idx, (data, target) in enumerate(loaders['valid']):
            # move to GPU
            if use_cuda:
                data, target = data.cuda(), target.cuda()
            # forward pass: compute predicted outputs by passing inputs to the model
            output = model(data)
            # calculate the batch loss
            loss = criterion(output, target)
            # update average validation loss 
            #valid_loss += loss.item()*data.size(0)
            valid_loss = valid_loss + ((1 / (batch_idx + 1)) * (loss.data - valid_loss))
            
        # print training/validation statistics 
        print('Epoch: {} \tTraining Loss: {:.6f} \tValidation Loss: {:.6f}'.format(
            epoch, 
            train_loss,
            valid_loss
            ))
        
        #save the model if validation loss has decreased
        if valid_loss <= valid_loss_min:
            print('Validation loss decreased ({:.6f} --> {:.6f}).  Saving model ...'.format(
            valid_loss_min,
            valid_loss))
            torch.save(model.state_dict(), save_path)
            valid_loss_min = valid_loss

    # return trained model
    return model

def test(loaders, model, criterion, use_cuda):

    # monitor test loss and accuracy
    test_loss = 0.
    correct = 0.
    total = 0.

    model.eval()
    for batch_idx, (data, target) in enumerate(loaders['test']):
        # move to GPU
        if use_cuda:
            data, target = data.cuda(), target.cuda()
        # forward pass: compute predicted outputs by passing inputs to the model
        output = model(data)
        # calculate the loss
        loss = criterion(output, target)
        # update average test loss 
        test_loss = test_loss + ((1 / (batch_idx + 1)) * (loss.data - test_loss))
        # convert output probabilities to predicted class
        pred = output.data.max(1, keepdim=True)[1]
        # compare predictions to true label
        correct += np.sum(np.squeeze(pred.eq(target.data.view_as(pred))).cpu().numpy())
        total += data.size(0)
            
    print('Test Loss: {:.6f}\n'.format(test_loss))

    print('\nTest Accuracy: %2d%% (%2d/%2d)' % (

Question 5: Outline the steps you took to get to your final CNN architecture and your reasoning at each step. Describe why you think the architecture is suitable for the current problem.

Answer:

The VGG16 pretrained model has two sections. The feature extracter and then the classifier. The fearture extracter has about 13 convolutional layers which is pretrained on the Imagenet database. Since this database has 1000 different classes including dog breeds, these feature extracters seem to be optimsed for dogs as well. In thi sproject we have a "small, but similar" database, as explained in the lessons. Hence leaving the parameters in feature extracter unchanged seems reasonable, since we want the network to extract similar features for our dog breed classifier. The last part, which is the classifier required small changes. The original model takes in 25088 parameters for the fully connected block and has output for the 1000 classes in the Imagenet database. This output number had to be changed to 133 since we want only a dog breed classifier with 133 different types. Hence only changing the final layer of the fully connected block was sufficient. This architecture seems appropriate since this model was already used on one of the largest databases and achieved some good results. Hence adapting this model to be a classifier of a smaller database containing classes which the model has previously dealt with seems reasonable.

# train the model

def train(n_epochs, loaders, model, optimizer, criterion, use_cuda, save_path):
    """returns trained model"""
    # initialize tracker for minimum validation loss
    valid_loss_min = np.Inf 
    
    for epoch in range(1, n_epochs+1):
        # initialize variables to monitor training and validation loss
        train_loss = 0.0
        valid_loss = 0.0
        
        ###################
        # train the model #
        ###################
        model.train()
        for batch_idx, (data, target) in enumerate(loaders['train']):
            # move to GPU
            if use_cuda:
                data, target = data.cuda(), target.cuda()
            ## find the loss and update the model parameters accordingly
            ## record the average training loss, using something like
            ## train_loss = train_loss + ((1 / (batch_idx + 1)) * (loss.data - train_loss))
            
            #zero the gradients
            optimizer.zero_grad()
            # forward pass
            output = model(data)
            # calculate the batch loss
            loss = criterion(output, target)
            # backward pass: compute gradient of the loss with respect to model parameters
            loss.backward()
            # perform a single optimization step (parameter update)
            optimizer.step()
            #Update train loss
            train_loss = train_loss + ((1 / (batch_idx + 1)) * (loss.data - train_loss))
            
        ######################    
        # validate the model #
        ######################
        model.eval()
        for batch_idx, (data, target) in enumerate(loaders['valid']):
            # move to GPU
            if use_cuda:
                data, target = data.cuda(), target.cuda()
            # forward pass: compute predicted outputs by passing inputs to the model
            output = model(data)
            # calculate the batch loss
            loss = criterion(output, target)
            # update average validation loss 
            #valid_loss += loss.item()*data.size(0)
            valid_loss = valid_loss + ((1 / (batch_idx + 1)) * (loss.data - valid_loss))
            
        # print training/validation statistics 
        print('Epoch: {} \tTraining Loss: {:.6f} \tValidation Loss: {:.6f}'.format(
            epoch, 
            train_loss,
            valid_loss
            ))
        
        #save the model if validation loss has decreased
        if valid_loss <= valid_loss_min:
            print('Validation loss decreased ({:.6f} --> {:.6f}).  Saving model ...'.format(
            valid_loss_min,
            valid_loss))
            torch.save(model.state_dict(), save_path)
            valid_loss_min = valid_loss

    # return trained model
    return model

def image_loader(loader, image_name):
    image = Image.open(image_name)
    image = loader(image).float()
    image = torch.tensor(image, requires_grad=False)
    image = image.unsqueeze(0)
    return image


def predict_breed_transfer(img_path):
    # load the image and return the predicted breed
    model_transfer.eval()
    
    ## Load and pre-process an image from the given img_path
    normalize = transforms.Normalize(mean=[0.485, 0.456, 0.406],
                                     std=[0.229, 0.224, 0.225])
    data_transforms = transforms.Compose([
        transforms.Resize(256),
        transforms.CenterCrop(224),
        transforms.ToTensor(),
        normalize,
    ])
    
    loaded_image = image_loader(data_transforms, img_path)
    
    if use_cuda:
        loaded_image = loaded_image.cuda()
    
    
    ## Return the *index* of the predicted class for that image
    idx = np.argmax(model_transfer(loaded_image).detach().cpu().numpy())
    
    return class_names[idx] # predicted class index
